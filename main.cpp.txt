#include <iostream>
#include <vector>
#include <string>
#include <fstream>

using namespace std;

// --- Classes Base ---
class Cidade {
public:
    string nome;
    Cidade(string n) : nome(n) {}
};

class Transporte {
public:
    string nome;
    int capacidade;
    Cidade* localAtual;
    Transporte(string n, int cap, Cidade* loc) : nome(n), capacidade(cap), localAtual(loc) {}
};

class Passageiro {
public:
    string nome;
    Cidade* localAtual;
    Passageiro(string n, Cidade* loc) : nome(n), localAtual(loc) {}
};

// --- Controlador Principal ---
class Controlador {
    vector<Cidade*> cidades;
    vector<Transporte*> transportes;
    vector<Passageiro*> passageiros;

public:
    void cadastrarCidade(string nome) { cidades.push_back(new Cidade(nome)); }
    
    void listarDados() {
        cout << "\n--- Cidades Cadastradas ---\n";
        for (auto c : cidades) cout << "- " << c->nome << endl;
    }

    void salvarDados() {
        ofstream file("dados.txt");
        for (auto c : cidades) file << "CIDADE:" << c->nome << endl;
        file.close();
        cout << "Dados salvos com sucesso em dados.txt\n";
    }
};

int main() {
    Controlador sistema;
    int opcao;

    do {
        cout << "\n1. Cadastrar Cidade\n2. Listar\n3. Salvar\n0. Sair\nEscolha: ";
        cin >> opcao;

        if (opcao == 1) {
            string nome;
            cout << "Nome da cidade: "; cin >> nome;
            sistema.cadastrarCidade(nome);
        } else if (opcao == 2) {
            sistema.listarDados();
        } else if (opcao == 3) {
            sistema.salvarDados();
        }
    } while (opcao != 0);

    return 0;
}
